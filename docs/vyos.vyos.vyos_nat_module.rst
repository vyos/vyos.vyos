Examples
--------

.. code-block:: yaml

    # Using merged
    - name: Merge NAT source rule
      vyos.vyos.vyos_nat:
        config:
          source:
            rule:
              - id: 100
                description: "Outbound masquerade"
        state: merged

    # Using gathered
    - name: Gather NAT config
      vyos.vyos.vyos_nat:
        state: gathered

    # Using deleted
    - name: Delete NAT config
      vyos.vyos.vyos_nat:
        state: deleted

    # Using replaced
    - name: Replace NAT config
      vyos.vyos.vyos_nat:
        config:
          source:
            rule:
              - id: 100
                description: "Replaced rule"
        state: replaced

    # Using parsed
    - name: Parse NAT config
      vyos.vyos.vyos_nat:
        running_config: "{{ lookup('file', './nat_config.cfg') }}"
        state: parsed

    # Using rendered
    - name: Render NAT config offline
      vyos.vyos.vyos_nat:
        config:
          source:
            rule:
              - id: 100
                description: "Rendered rule"
        state: rendered
